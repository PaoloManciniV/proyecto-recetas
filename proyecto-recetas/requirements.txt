# Dependencias del proyecto. Se instalan con:  pip install -r requirements.txt

fastapi              # framework para construir la API
uvicorn[standard]    # servidor que corre la app de FastAPI
sqlalchemy           # para trabajar con la base de datos
pydantic             # validación de datos (lo usa FastAPI)
email-validator      # valida que los correos sean válidos (EmailStr)
python-dotenv        # para leer el archivo .env
bcrypt               # para encriptar contraseñas
pyjwt                # para crear y leer los tokens JWT (login)
pytest               # para correr las pruebas unitarias
httpx                # lo necesita el TestClient de las pruebas
pymysql              # driver de MySQL (lo usarán al desplegar con Docker)
